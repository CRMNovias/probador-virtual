// Componente: Galería de Pruebas Virtuales
const GalleryScreen = ({ generatedImages, onDeleteImage, setModalImage, setShareModalImage }) => {
    const [expandedDress, setExpandedDress] = useState(null);
    const [deleteConfirmImage, setDeleteConfirmImage] = useState(null);

    const imagesByDress = generatedImages.reduce((acc, img) => {
        if (!acc[img.dressId]) { acc[img.dressId] = []; }
        acc[img.dressId].push(img);
        return acc;
    }, {});
    
    const toggleExpand = (dressId) => {
      setExpandedDress(prev => (prev === dressId ? null : dressId));
    };

    const handleDelete = (e, img) => {
        e.stopPropagation();
        setDeleteConfirmImage(img);
    };
    
    const handleShare = (e, img) => {
        e.stopPropagation();
        setShareModalImage(img);
    }
    
    const confirmDelete = () => {
        onDeleteImage(deleteConfirmImage.id);
        setDeleteConfirmImage(null);
    }

    return (
        <div className="w-full max-w-5xl mx-auto p-4 md:p-8">
            <h1 className="text-4xl font-serif text-[#4a3f35] mb-8 text-center">Mi Galería</h1>
            
            {Object.keys(imagesByDress).length === 0 ? (
                <p className="text-center text-gray-500 font-light mt-12">Aún no has generado ninguna prueba. ¡Ve al probador para empezar!</p>
            ) : (
                <div className="space-y-4">
                    {Object.entries(imagesByDress).map(([dressId, images]) => {
                        const dress = DRESSES_DATA[dressId];
                        const isExpanded = expandedDress === dressId;
                        return (
                            <div key={dressId} className="bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden">
                                <button onClick={() => toggleExpand(dressId)} className="w-full flex justify-between items-center p-4 text-left">
                                    <div>
                                        <h2 className="text-xl font-serif text-[#4a3f35]">{dress.name}</h2>
                                        <p className="text-sm text-gray-500 font-light">{images.length} prueba(s) generada(s)</p>
                                    </div>
                                    <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={`transform transition-transform duration-300 ${isExpanded ? 'rotate-180' : ''}`}><polyline points="6 9 12 15 18 9"></polyline></svg>
                                </button>
                                {isExpanded && (
                                    <div className="p-4 bg-gray-50 border-t border-gray-200">
                                        <div className="grid grid-cols-2 sm:grid-cols-3 md:grid-cols-4 gap-4">
                                            {images.map(img => (
                                                <div key={img.id} className="group relative rounded-lg overflow-hidden aspect-[2/3] cursor-pointer" onClick={() => setModalImage(img.url)}>
                                                    <img src={img.url} alt={`Prueba ${img.id}`} className="w-full h-full object-cover" />
                                                    <div className="absolute inset-0 bg-black/40 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center gap-3">
                                                         <button onClick={(e) => { e.stopPropagation(); setModalImage(img.url)}} className="p-2 bg-white/80 rounded-full text-gray-800 hover:scale-110 transition-transform"><MaximizeIcon/></button>
                                                         <button onClick={(e) => handleShare(e, img)} className="p-2 bg-white/80 rounded-full text-gray-800 hover:scale-110 transition-transform"><ShareIcon/></button>
                                                         <button onClick={(e) => handleDelete(e, img)} className="p-2 bg-white/80 rounded-full text-red-500 hover:scale-110 transition-transform"><TrashIcon/></button>
                                                    </div>
                                                </div>
                                            ))}
                                        </div>
                                    </div>
                                )}
                            </div>
                        );
                    })}
                </div>
            )}
            
            {deleteConfirmImage && <DeleteConfirmationModal onConfirm={confirmDelete} onCancel={() => setDeleteConfirmImage(null)} />}
        </div>
    );
};

// --- Componente Principal de la Aplicación ---
export default function App() {
    const [currentView, setCurrentView] = useState('auth-phone');
    const [isLoading, setIsLoading] = useState(false);
    const [avatarUrl, setAvatarUrl] = useState(null);
    const [generatedImages, setGeneratedImages] = useState([]);
    const [modalImage, setModalImage] = useState(null);
    const [shareModalImage, setShareModalImage] = useState(null);
    
    const handlePhoneSubmit = (phone) => {
        setIsLoading(true);
        setTimeout(() => { setIsLoading(false); setCurrentView('auth-code'); }, 1500);
    };

    const handleCodeVerify = (code) => {
        setIsLoading(true);
        setTimeout(() => { setIsLoading(false); setCurrentView('avatar-upload'); }, 1500);
    };
    
    const handlePhotoUpload = (file) => {
        setCurrentView('avatar-generating');
        setTimeout(() => {
            setAvatarUrl('https://placehold.co/800x1200/FDFBFA/333?text=Tu+Avatar');
            setCurrentView('try-on');
        }, 3000);
    };
    
    const handleResetAvatar = (regenerate) => {
        if (regenerate) {
             setCurrentView('avatar-generating');
             setTimeout(() => {
                setAvatarUrl(`https://placehold.co/800x1200/FDFBFA/333?text=Avatar+Nuevo`);
                setCurrentView('try-on');
            }, 3000);
        } else {
            setCurrentView('avatar-upload');
        }
    };

    const handleGenerateTryOn = (newImage) => {
        setGeneratedImages(prev => [newImage, ...prev]);
    };
    
    const handleDeleteImage = (imageId) => {
        setGeneratedImages(prev => prev.filter(img => img.id !== imageId));
    };
    
    const renderContent = () => {
        if (isLoading) return <Loader text="Procesando..." />;
        switch (currentView) {
            case 'auth-phone': return <AuthPhoneScreen onPhoneSubmit={handlePhoneSubmit} />;
            case 'auth-code': return <AuthCodeScreen onCodeVerify={handleCodeVerify} />;
            case 'avatar-upload': return <AvatarUploadScreen onPhotoUpload={handlePhotoUpload} />;
            case 'avatar-generating': return <Loader text="Nuestra IA está creando tu avatar..." />;
            case 'try-on': return <TryOnScreen 
                avatarUrl={avatarUrl} 
                onGenerate={handleGenerateTryOn} 
                onResetAvatar={handleResetAvatar} 
                onDeleteGeneratedImage={handleDeleteImage}
                setModalImage={setModalImage}
                setShareModalImage={setShareModalImage}
            />;
            case 'gallery': return <GalleryScreen 
                generatedImages={generatedImages} 
                onDeleteImage={handleDeleteImage} 
                setModalImage={setModalImage}
                setShareModalImage={setShareModalImage}
            />;
            case 'appointments': return <AppointmentsScreen />;
            default: return <AuthPhoneScreen onPhoneSubmit={handlePhoneSubmit} />;
        }
    };

    const showMainApp = !['auth-phone', 'auth-code', 'avatar-upload', 'avatar-generating'].includes(currentView);

    return (
        <div className="font-sans antialiased bg-[#F8F7F5] min-h-screen text-[#4a3f35]">
            <header className="sticky top-0 bg-white/80 backdrop-blur-md z-10 shadow-sm">
                <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 flex justify-between items-center h-20">
                    <h1 className="text-2xl font-serif tracking-wider">Atelier de Bodas</h1>
                    <button onClick={() => alert('Cerrando aplicación...')} className="p-2 rounded-full hover:bg-gray-200 transition-colors">
                        <XIcon />
                    </button>
                </div>
            </header>

            <main className="flex items-center justify-center pt-8 pb-48">
                 {renderContent()}
            </main>

            {showMainApp && (
                 <>
                    <div className="fixed bottom-24 left-1/2 -translate-x-1/2 z-20">
                         <button onClick={() => alert('Cerrando app y abriendo calendario de reservas...')} className="text-white bg-[#8C6F5A] rounded-full px-8 py-4 shadow-xl hover:bg-[#7a5f4d] transition-all transform hover:scale-105 flex items-center gap-3 text-lg font-semibold">
                            <CalendarIcon />
                            Reservar Cita
                         </button>
                    </div>
                    <div className="fixed bottom-0 left-0 right-0 z-10 bg-white/80 backdrop-blur-lg border-t border-gray-200">
                        <div className="max-w-md mx-auto flex justify-around items-center p-2">
                             <NavButton icon={<CameraIcon />} label="Probador" active={currentView === 'try-on'} onClick={() => setCurrentView('try-on')} />
                             <NavButton icon={<ImageIcon />} label="Galería" active={currentView === 'gallery'} onClick={() => setCurrentView('gallery')} />
                             <NavButton icon={<HistoryIcon />} label="Citas" active={currentView === 'appointments'} onClick={() => setCurrentView('appointments')} />
                        </div>
                    </div>
                 </>
            )}

            {/* Modales Globales */}
            {modalImage && <ImageViewerModal url={modalImage} onClose={() => setModalImage(null)} />}
            {shareModalImage && <ShareModal image={shareModalImage} onClose={() => setShareModalImage(null)} />}

        </div>
    );
}

const NavButton = ({ icon, label, active, onClick }) => (
    <button
        onClick={onClick}
        className={`flex flex-col items-center justify-center w-24 h-16 rounded-lg transition-colors ${active ? 'text-[#8C6F5A]' : 'text-gray-500 hover:text-[#8C6F5A]'}`}
    >
        <div className="w-8 h-8">{icon}</div>
        <span className="text-xs mt-1 font-medium">{label}</span>
    </button>
);
